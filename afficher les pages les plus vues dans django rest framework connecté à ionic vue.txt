Pour afficher les pages les plus vues dans Django Rest Framework, vous pouvez suivre les étapes suivantes :
Créer un modèle pour stocker les informations des pages vues :
from django.db import models

class PageView(models.Model):
    page = models.CharField(max_length=100)
    count = models.IntegerField(default=0)
Créer un sérialiseur pour sérialiser les données des pages vues :
from rest_framework import serializers
from .models import PageView

class PageViewSerializer(serializers.ModelSerializer):
    class Meta:
        model = PageView
        fields = ('page', 'count')
Créer une vue pour afficher les pages les plus vues :
from rest_framework.views import APIView
from rest_framework.response import Response
from .models import PageView
from .serializers import PageViewSerializer

class MostViewedPages(APIView):
    def get(self, request):
        most_viewed_pages = PageView.objects.order_by('-count')[:10]  # Récupérer les 10 pages les plus vues
        serializer = PageViewSerializer(most_viewed_pages, many=True)
        return Response(serializer.data)
Configurer l'URL pour à la vue :
from django.urls import path
from .views import MostViewedPages

urlpatterns = [
    path('most-viewed-pages/', MostViewedPages.as_view(), name='most_viewed_pages'),
] Maintenant, vous pouvez à la liste des pages les plus vues en utilisant l'URL /most-viewed-pages/.

Dans Django :
Créez un PageViewmodèle dansmodels.py :
class PageView(models.Model):
    url = models.URLField()
    view_count = models.IntegerField(default=0)
    last_viewed = models.DateTimeField(auto_now=True)
Créez une vue API pour obtenir les pages les plus consultées dansviews.py :
class MostViewedPagesView(APIView):
    def get(self, request):
        most_viewed_pages = PageView.objects.order_by('-view_count')[:10]
        serializer = PageViewSerializer(most_viewed_pages, many=True)
        return Response(serializer.data)
